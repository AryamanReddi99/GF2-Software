Devices {
   A is XOR;
   B is AND;
   C is NAND;
   D is OR;
   E is NOR;
   F is DTYPE;
   G is NAND;
   H is OR;
   A,B,C,D,G,H have 2 inputs;
   E has 3 inputs;
   S1 => S3 are SWITCH;
   S2 starts high;
   CKL1 is CLOCK;
   CKL1 has cycle 5;
}

Connections {
   Device A {
       S1 to A.1;
       S2 to A.2;
   }
   Device B {
       S1 to B.1;
       S3 to B.2;
   }
   Device C {
       S2 to C.1;
       B to C.2;
   }
   Device D {
       A to D.1;
       B to D.2;
   }
   Device E {
       D to E.1;
       S2 to E.2;
       C to E.3;
   }
   Device F {
        CLK1 to F.CLK;
        D to F.SET;
        C to F.CLEAR;
        S2 to F.DATA;
   }
   Device G {
       E to G.1;
       F to G.2;
   }
   Device H {
       C to H.1;
       F.QBAR to H.2;
   }
}

Monitor {
   B, G, H;
}
