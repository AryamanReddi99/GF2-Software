Devices: {
    A is an XOR gate
    B is a AND gate
    C is a NAND gate
    D is an OR gate
    E is NOR gate
    F is a DTYPE
    G is NAND
    H is OR
    A,B,C,D,G,H have 2 inputs
    E has 3 inputs
}

Monitor: {
    B, G, H
} 

Init: {
    S1 => S3 are SWITCH inputs
    set S2 high
    CKL1 is a CLOCK input
    CKL1 has cycle time of 5
}

Connections: {
    Device A: {
        S1 to A.1
        S2 to A.2
    }
    Device B: {
        S1 to B.1
        S3 to B.2
    }
    Device C: {
        S2 to C.1
        B to C.2
    }
    Device D: {
        A to D.1
        B to D.2
    }
    Device E: {
        D to E.1
        S2 to E.2
        C to E.3
    }
    Device F: {
        S2 to F.1
        C to F.2
        D to F.3
        CLK1 to F.4
    }
    Device G: {
        E to G.1
        F to G.2
    }
    Device H: {
        C to H.1
        not F to H.2
    }
}
